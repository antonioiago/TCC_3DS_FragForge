create database if not exists fragforge;
use fragforge;

create table funcao
(
	id_funcao int not null primary key,
    nome_funcao varchar(20) not null,
    icon_funcao mediumblob
);

create table patente
(
	id_patente int not null primary key,
    nome_patente varchar(20) not null,
    icon_patente mediumblob
);

create table jogador
(
	id_jogador int auto_increment primary key,
	nickname_jogador varchar(100) not null,
    codigo_battlenet varchar(100) not null unique,
    email_jogador varchar(100) not null unique,
    senha_jogador varchar(100),
    id_funcao int,
    id_patente int,
    foto_jogador mediumblob,
    pontuacao_jogador int,
	id_equipe int,
    foreign key (id_funcao) references funcao (id_funcao),
    foreign key (id_patente) references patente (id_patente),
	foreign key (id_equipe) references equipe (id_equipe)
);

create table highlight
(
	id_highlight int auto_increment primary key,
    id_jogador int not null,
    print_estatistica mediumblob,
    jogada longblob,
	foreign key (id_jogador) references jogador(id_jogador)
);

create table adm
(
	id_adm int auto_increment primary key,
    nickname_adm varchar(100) not null,
    email_adm varchar(100) not null unique,
    senha_adm varchar(100) not null,
    foto_adm mediumblob
);

create table equipe
(
	id_equipe int auto_increment primary key,
    nome_equipe varchar(100) not null unique,
    foto_equipe mediumblob,
    pontuacao_equipe int
);

select * from jogador;
select * from highlight;
select * from adm;
select * from equipe;
select * from funcao;
