create database Talvez;
use Talvez;


CREATE TABLE usuarios (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(100),
  email VARCHAR(100) UNIQUE,
  senha VARCHAR(255),
  tipo ENUM('tecnico','empresa')
);

CREATE TABLE vagas (
  id INT AUTO_INCREMENT PRIMARY KEY,
  titulo VARCHAR(150),
  descricao TEXT,
  empresa_id INT,
  FOREIGN KEY (empresa_id) REFERENCES usuarios(id)
);
